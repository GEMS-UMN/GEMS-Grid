31def grid_id_to_corner_coord(gid, level, shift=False): 

32 """Convert GEMS Grid ID into an EASE v2 corner coordinate. 

33 

34 Parameters 

35 ---------- 

36 gid : str 

37 Grid ID of the cell. 

38 corner : str 

39 The type of corner to generate coordinates for 

40 (upper_left | upper_right | lower_right | lower_left) 

41 Returns 

42 ---------- 

43 corner_x, corner_y : np.array 

44 EASE grid coordinate (x, y) of the cell' corner. Resolution determined using grid ID. 

45 """ 

46 # r_digit = 6 

47 # decompse the GEMS grid ID into its level parts. Then remove the first compoenent 

48 # which, is simply represent the Level of the cell. 

49 id_elements = gid.split('.') 

50 # level = int(id_elements[0][-1]) 

51 id_elements.pop(0) 

52 

53 # the gridID contains the index values for number of rows|columns, from a 0,0 orgin 

54 # This is the 'global' position within the larger grid 

55 # 

56 # Remember format is Lx.RRRCCC.RC.RC ...; y=R, x=C 

57 # 

58 # Those R, C components indicate the location/number of grid cells that 

59 # correspond with the location in gems grid space 

60 # 

61 # This can be explicitly exploited such that: 

62 # ([RRR, R, ... R] * [level[0]['x_length], ... level[6]['x_length]]).sum() 

63 # ([CCC, C, ... C] * [level[0]['y_length],... level[6]['y_length]]).sum() 

64 # will give the coordinates to the edge of the pixel. For example: 

65 # 

66 # [L6.202482.13.21.00.00.00.00] would decompose into 

67 # x = [482, 3, 2, 0, 0, 0, 0] 

68 # y = [202, 1, 1, 0, 0, 0, 0] 

69 # 

70 # To determining the coordinate of the left, top edges at Level 2 

71 # x = 482 * level[0]['x_length] + 3 * [level[1]['x_length] + 2 * [level[`2`]['x_length] 

72 # y = 202 * [level[0]['x_length] + 1 * [level[1]['x_length] + 1 * [level[2]['x_length] 

73 # 

74 # and to determin if it is an edge at level 2, all the remaining Level components must be 

75 # zero: [ 0, 0, 0, 0]] summing them all ensure that they are zero 

76 # 

77 # if we asked for the Level 1 coordinate, it would be different. x|y coord is calculated 

78 # using the L0, L1 compoents: 

79 # x = 482 * level[0]['x_length] + 3 * [level[1]['x_length] + 2 

80 # 

81 # but then we need to check if it is on an L1 edge. this requires all other compontes to be 

82 # zeros: 

83 # [2, 0, 0, 0, 0] 

84 # since it is not a Level 1 edge, if we needed a right, bottom value, we'd have to shfit 

85 # the last L1 index up by one 

86 

87 # beyond level 0, the row | column index is grid location within the L0 pixes 

88 # This is the relative position within the gloabl position. 

89 # spliting these into indexes, which requires converstion from str to ints 

90 level_x_i = np.array([int(id[1]) if len(id) != 6 else int(id[3:]) \ 

91 for id in id_elements], dtype=int) 

92 level_y_i = np.array([int(id[0]) if len(id) != 6 else int(id[0:3]) \ 

93 for id in id_elements], dtype=int) 

94 

95 # remainders of levels are useful for figuring out if on edge on not 

96 # depending on supplied level, we can drop unnecesary levels for 

97 # determining the coodinates, but only need the levels < specified 

98 # level for determing if it is an edge pixel or not 

99 if int(level) < 6: 

100 level_x_r = level_x_i[int(level) + 1:] 

101 level_y_r = level_y_i[int(level) + 1:] 

102 else: 

103 level_x_r = level_x_i[int(level):] 

104 level_y_r = level_y_i[int(level):] 

105 

106 level_x_i = level_x_i[:int(level) + 1] 

107 level_y_i = level_y_i[:int(level) + 1] 

108 

109 if (level_x_r.sum() != 0) and shift: 

110 # print(f'level_x_r: {level_x_r}; sum: {level_x_r.sum()}') 

111 # print('shifting x') 

112 level_x_i[-1] += 1 

113 

114 if (level_y_r.sum() != 0) and shift: 

115 # print(f'level_y_r: {level_y_r}; sum: {level_y_r.sum()}') 

116 # print('shifting y') 

117 level_y_i[-1] += 1 

118 

119 # print (f'level_x_i: {level_x_i}; level_y_i: {level_y_i}') 

120 corner_x = np.array([level_x_i[i] * levels_specs[i]['x_length'] \ 

121 for i in range(level_x_i.shape[0])]).sum() 

122 

123 corner_y = np.array([level_y_i[i] * levels_specs[i]['y_length'] \ 

124 for i in range(level_y_i.shape[0])]).sum() 

125 

126 return corner_x, corner_y

136def gems_grid_bounds(bounds, source_crs, level): 

137 """Determine EASE v2 coords corresponding to bounds of GEMS Grid 

138 

139 Parameters 

140 ---------- 

141 bounds : tuple 

142 Tuple with the (min_x/left, min_y/bottom, max_x/right, max_y/top) of the soure raster. 

143 level : int 

144 GEMS Grid Level of the 

145 source_crs : int 

146 EPSG code of coordinate reference systems for the source raster. 

147 

148 Returns 

149 ---------- 

150 gems_grid_bounds : tuple of floats 

151 Bounding box coordidinates with valid GEMS Grid coordinates. 

152 """ 

153 # first, convert the bounds coordinates into corner coordinates. these are tuples. 

154 # next, convert the corners to 'line_segements'. these are the coordinate pairs 

155 # connecting the corners 

156 # 

157 # Those line segmensts have 'duplicate' entries, since each line segment ends where next starts 

158 # once we've delt with non-ease sourc_src below, we'll drop the final element of 

159 # of line_segments, to reduce duplicates 

160 lower_left, upper_left, lower_right, upper_right = \ 

161 chain(product([bounds[0], bounds[2]], [bounds[1], bounds[3]])) 

162 # print(f'conrner passed in: {corners}') 

163 line_segments = list(pairwise_circle([upper_left, upper_right, lower_right, lower_left])) 

164 

165 # if not in EASE v2, need to build a cood tranformer to get coords into EASE 

166 # also need to add nodes to the each line segment. once that is done, we use 

167 # the transformer to convert all individual nodes to EASE, while maintaining the 

168 # original order. 

169 # then use 

170 if source_crs != ease_crs: 

171 line_segments = [add_nodes(ls[0], ls[1]) for ls in line_segments] 

172 tranform_coords = Transformer.from_crs(source_crs, ease_crs, always_xy=True).transform 

173 line_segments = [[tranform_coords(c[0], c[1]) for c in seg] for seg in line_segments] 

174 

175 # may not need to drop the last elements in each list (recall, the last element of 

176 # one list, is the start of the next) since we'll just be using the nodes as Points 

177 # in a geoseries. but if we wanted to make Polygon, the duplciate values pose 

178 # problems 

179 line_segments = flatten([seg[:-1] for seg in line_segments]) 

180 bound_box = GeoSeries([Point(c[0], c[1]) for c in line_segments], 

181 crs=ease_crs) 

182 bound_box = bound_box.total_bounds 

183 

184 # now that we have ensured we have accounted for bounds not necessarily 

185 # corresponding to vertices in EASE coordinates, we'll create the 

186 # bounding box as vertices 

187 bound_box = GeoSeries([Point(bound_box[0], bound_box[3]), # upper_left 

188 Point(bound_box[2], bound_box[1])], # lower_right 

189 crs=ease_crs) 

190 

191 bbox_grid_xy = coords_ease_to_coords_grid(bound_box) 

192 bbox_gridID = bbox_grid_xy.apply(lambda x: _grid_xy_to_grid_id(x, 6)) 

193 

194 # if we use upper_right, and lower left coords only, we can apply 

195 # shift or not, to both points. 

196 upper_left = grid_id_to_corner_coord(bbox_gridID.iloc[0], level=level) 

197 lower_right = grid_id_to_corner_coord(bbox_gridID.iloc[1], level=level, shift=True) 

198 

199 ease_xs = [upper_left[0], lower_right[0]] 

200 ease_ys = [lower_right[1], upper_left[1]] 

201 

202 ease_xs = np.array( 

203 [shift_range_grid_multiple(ease_xs[i], axis='x') \ 

204 for i in range(len(ease_xs))] 

205 ) 

206 ease_ys = np.array( 

207 [shift_range_grid_multiple(ease_ys[i], axis='y') \ 

208 for i in range(len(ease_ys))] 

209 ) 

210 

211 return [ease_xs[0], ease_ys[0], ease_xs[1], ease_ys[1]] # , decimals=8)

214def gems_align_check(in_raster, gems_level): 

215 """Check if a raster is aligned to the GEMS grid. 

216 

217 Takes a path to a raster and performs 6 tests of alignment with the 

218 GEMS grid. The first 4 tests ensure the grid is defined the same as 

219 the GEMS grid. If a raster is the same projection (EPSG:6933), is 

220 rectilinear (aligned to the axes), has square cells, and is one of 

221 the 7 GEMS resolutions, it is able to be aligned to the GEMS grid. 

222 The final 2 tests check that if you were to extrapolate your 

223 raster out to a global extent, that the upper left corner of the 

224 GEMS grid falls on the upper left corner of one of the input 

225 raster's extraploted cells. 

226 

227 

228 Parameters 

229 ------------ 

230 in_raster: str, file object or pathlib.Path object 

231 Filename and path of input raster. 

232 

233 gems_level: int 

234 Integer from 0 to 6 corresponding to the GEMS grid levels 

235 (i.e., resolutions). 

236 

237 Returns 

238 ------- 

239 results: Dictionary 

240 Dictionary of test name and result of test (pass/fail). 

241 

242 """ 

243 if gems_level not in [0, 1, 2, 3, 4, 5, 6]: 

244 raise Exception("Invalid grid level. Options are: 0, 1, 2, 3, 4, 5, 6") 

245 

246 # Calculate the geotransform of the GEMS grid 

247 gems_t = rasterio.transform.from_origin( 

248 grid_spec['ease']['min_x'], 

249 grid_spec['ease']['max_y'], 

250 levels_specs[gems_level]['x_length'], 

251 levels_specs[gems_level]['x_length'] 

252 ) 

253 

254 with rasterio.open(in_raster, 'r') as src: 

255 src_t = src.transform 

256 test_count = 6 

257 

258 results = { 

259 'Projection is EASE-Grid 2.0 (EPSG:6933):': 'Pass', 

260 'Dataset is rectilinear:': 'Pass', 

261 'X and Y size are equal:': 'Pass', 

262 'Cell size matches a GEMS grid resolution:': 'Pass', 

263 'The X dimension of the cell corners matches a GEMS grid corner:': 'Pass', 

264 'The Y dimension of the cell corners matches a GEMS grid corner:': 'Pass', 

265 'Tests passed out of 6:': test_count 

266 } 

267 

268 if src.crs != 'EPSG: 6933': 

269 results['Projection is EASE-Grid 2.0 (EPSG:6933):'] = str('Fail. Supplied CRS is ' + str(src.crs)) 

270 test_count -= 1 

271 

272 if not src_t.is_rectilinear: 

273 results['Dataset is rectilinear:'] = 'Fail' 

274 test_count -= 1 

275 

276 if not epsilon_check(abs(src_t.a), abs(src_t.e)): 

277 results['X and Y size are equal:'] = 'Fail' 

278 test_count -= 1 

279 

280 if not epsilon_check(gems_t.a, src_t.a): 

281 results['Cell size matches a GEMS grid resolution:'] = 'Fail' 

282 test_count -= 1 

283 

284 # Find the input raster coordinates of the hypothetical input raster cell that overlaps the GEMS origin 

285 xr, yr = ~src_t * (gems_t.xoff, gems_t.yoff) 

286 

287 # Move to the coordinates of the upper left corner of the above cell 

288 xrnd = math.floor(xr + 0.5) if xr > 0.0 else math.ceil(xr - 0.5) 

289 if epsilon_check(xrnd, xr): # Test if on an edge, else go left 

290 xr = float(xrnd) 

291 else: 

292 xr = float(math.floor(xr)) 

293 

294 yrnd = math.floor(yr + 0.5) if yr > 0.0 else math.ceil(yr - 0.5) 

295 if epsilon_check(yrnd, yr): # Test if on an edge, else go up 

296 yr = float(yrnd) 

297 else: 

298 yr = float(math.floor(yr)) 

299 

300 # Convert from input raster coordinates to world (ease) coordinates 

301 xw, yw = src_t * (xr, yr) 

302 

303 # Compare x and y values calculated in previous step to gems origin coordinates 

304 if not epsilon_check(xw, gems_t.xoff): 

305 results['The X dimension of the cell corners matches a GEMS grid corner:'] = \ 

306 str('Fail. Difference = ' + str(round(abs(xw - gems_t.xoff), 5))) 

307 test_count -= 1 

308 

309 if not epsilon_check(yw, gems_t.yoff): 

310 results['The Y dimension of the cell corners matches a GEMS grid corner:'] = \ 

311 str('Fail. Difference = ' + str(round(abs(yw - gems_t.yoff), 5))) 

312 test_count -= 1 

313 

314 else: 

315 pass 

316 results['Tests passed out of 6:'] = str(test_count) 

317 

318 # for item in results: 

319 # print(item, results[item]) 

320 

321 return results
